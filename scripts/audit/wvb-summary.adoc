= WVB Source Summary Audit
:toc: left

This small audit utility scans canonical `.wvb` sources and writes a Markdown
summary of the Rust files, module declarations, and prose-listed file references
it can see.

It is not part of the product runtime.  It belongs in `scripts/audit/` because
it is a repository-maintenance inspection tool: useful while reducing
complexity, but not a top-level project artifact.

== Generated Files

`check_wvb.py` is the executable audit script.

`wvb_summary.md` is an intentionally committed snapshot report.  Re-run the
script after broad literate-source reshaping if the snapshot is useful for a
handoff or review.

[source,python]
----
// <<@file audit/check_wvb.py>>=
#!/usr/bin/env python3
"""Summarize generated files and module declarations in WVB sources."""

from __future__ import annotations

import argparse
import re
from pathlib import Path


DEFAULT_ROOTS = ("crates", "docs")


def project_root() -> Path:
    return Path(__file__).resolve().parents[2]


def parse_args() -> argparse.Namespace:
    parser = argparse.ArgumentParser(description=__doc__)
    parser.add_argument(
        "--root",
        type=Path,
        default=project_root(),
        help="project root to scan; default: repository root inferred from script path",
    )
    parser.add_argument(
        "--roots",
        nargs="*",
        default=list(DEFAULT_ROOTS),
        help="top-level directories to scan for .wvb files",
    )
    parser.add_argument(
        "--output",
        type=Path,
        default=Path("scripts/audit/wvb_summary.md"),
        help="output Markdown path, relative to --root unless absolute",
    )
    return parser.parse_args()


def rel(path: Path, root: Path) -> str:
    return path.relative_to(root).as_posix()


def discover_wvb_files(root: Path, scan_roots: list[str]) -> list[Path]:
    files: list[Path] = []
    for scan_root in scan_roots:
        base = root / scan_root
        if base.exists():
            files.extend(base.rglob("*.wvb"))
    return sorted(files)


def summarize_file(path: Path, root: Path) -> list[str]:
    content = path.read_text(encoding="utf-8")
    rust_files = re.findall(r"¤rust_file\(([^,]+)", content)
    mods = re.findall(r"^mod\s+([a-zA-Z0-9_]+);", content, re.MULTILINE)
    prose_refs = re.findall(r"^\*\s+`([^`]+)`", content, re.MULTILINE)

    if not rust_files and not mods and not prose_refs:
        return []

    lines = [f"## {rel(path, root)}"]
    if rust_files:
        lines.append(f"- rust_files: {', '.join(rust_files)}")
    if mods:
        lines.append(f"- mods: {', '.join(mods)}")
    if prose_refs:
        lines.append(f"- lists (prose): {', '.join(prose_refs)}")
    lines.append("")
    return lines


def main() -> int:
    args = parse_args()
    root = args.root.resolve()
    output = args.output if args.output.is_absolute() else root / args.output
    output.parent.mkdir(parents=True, exist_ok=True)

    lines: list[str] = [
        "# WVB Source Summary",
        "",
        "Generated by `scripts/audit/check_wvb.py`.",
        "",
    ]
    for path in discover_wvb_files(root, args.roots):
        try:
            lines.extend(summarize_file(path, root))
        except OSError as exc:
            lines.extend([f"## {rel(path, root)}", f"- error: {exc}", ""])

    output.write_text("\n".join(lines), encoding="utf-8")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
// @
