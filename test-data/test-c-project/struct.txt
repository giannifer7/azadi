    <[%(name) props]>

%b})

<[%(unit) snippets]>=

    return (%(name)Ptr) malloc(sizeof(%(name)));
%%})


%%pubmeth(%(name), err_t, init, %%{Allocator* allo%%}, %%{
    memset(self, 0, sizeof(%(name)));
    /*** %(name)_init example pre actions ***/
    /*** %(name)_init example actions ***/

    return 0;
cleanup:
    /*** %(name)_init example post actions ***/
    return -1;
%%})


%%pubmeth(%(name), err_t, uninit, Allocator* allo, %%{
    /*** %(name)_uninit example actions ***/
    return 0;
%%})

    %(name)_uninit(self, allo);
    FREE(allo, self, sizeof(%(name)));
    return 0;
%%})
