= lexer.c
:description: Literate source for the azadi macro lexer (C implementation)

This document is the literate source for `lexer.c`, the C implementation of
the azadi macro lexer.  The structure mirrors the Rust implementation in
`crates/azadi-macros/src/lexer/mod.rs`.

The lexer scans a buffer for a configurable _special character_ (default `%`)
and produces a flat array of tokens.  It is driven by a small explicit state
stack: `sBlock`, `sMacro`, and `sComment`.

Error handling follows the Rust version: errors are accumulated in a
caller-supplied `errors` array rather than causing an abort.  At EOF any
dangling special character, any unclosed `%/*` comment, and any unrecognised
`%X` sequence are all reported as errors and the offending text is emitted as a
`kText` token so that the output remains well-formed.

== Character-classification macros

These match the Rust predicates `is_identifier_start`, `is_identifier_continue`,
and `is_whitespace`.  They assume a local variable `c` of type `char`.

# <<character classification macros>>=
#define c_is_alpha  ((c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z'))
#define c_is_digit  (c >= '0' && c <= '9')
#define c_is_extra  (c == '_')
#define c_is_ident_start  (c_is_alpha || c_is_extra)
#define c_is_ident_cont   (c_is_alpha || c_is_digit || c_is_extra)
#define c_is_space  (c == ' ' || c == '\t' || c == '\r' || c == '\n')
# @

== Token and error types

`Token` carries a kind tag, source-file index, byte position within the
buffer, and byte length.  `LexerError` records the row/column and a
human-readable message, matching `types::LexerError` in Rust.

# <<token and error types>>=
typedef enum {
    kText        = 0,
    kSpace       = 1,
    kSpecial     = 2,
    kBlockOpen   = 3,
    kBlockClose  = 4,
    kMacro       = 5,
    kVar         = 6,
    kIdent       = 7,
    kComma       = 8,
    kCloseParen  = 9,
    kEqual       = 10,
    kLineComment = 11,
    kCommentOpen = 12,
    kCommentClose= 13,
    kEOF         = 14,
    /* legacy aliases kept for callers that pre-date the rename */
    kOther   = 0,   /* == kText  */
    kEndBuf  = 14,  /* == kEOF   */
} TokenType;

typedef struct {
    TokenType kind;
    int64_t   src;
    int64_t   pos;
    int64_t   leng;
} Token;

/* Maximum length of a lexer error message (including NUL). */
#define LEXER_ERROR_MSG_MAX 256

typedef struct {
    int64_t row;
    int64_t col;
    char    message[LEXER_ERROR_MSG_MAX];
} LexerError;
# @

        case kEndTokenBuf:        return "EndTokenBuf";

== String helpers

`snprintf`-based helper for recording errors into the caller-supplied array.

# <<string helpers>>=
/* Append an error entry.  Silently drops errors that overflow the array. */
static void push_error(LexerError *errors, int64_t *errors_idx,
                       int64_t errors_size,
                       int64_t row, int64_t col,
                       const char *fmt, ...)
{
    if (*errors_idx >= errors_size) return;
    LexerError *e = errors + (*errors_idx)++;
    e->row = row;
    e->col = col;
    va_list ap;
    va_start(ap, fmt);
    vsnprintf(e->message, LEXER_ERROR_MSG_MAX, fmt, ap);
    va_end(ap);
}
# @

== Internal macros

# <<lexer state macros>>=
#define in_state(xxst)  (states[*states_idx] == s##xxst)

#define PUSH_STATE(newState) do { \
    if (*states_idx >= states_size - 1) { return kEndStateStack; } \
    states[++(*states_idx)] = (newState); \
} while (0)

/* POP_STATE no longer returns kStateStackUnderrun on underrun; instead
   the caller has already emitted an error and will emit kText.
   A pop below depth 0 is a programming error; guard defensively. */
#define POP_STATE() do { \
    if (*states_idx == 0) { return kStateStackUnderrun; } \
    --(*states_idx); \
} while (0)

/* Advance la to the character after the next newline (or to end). */
#define ADVANCE_TO_EOL() \
    for (; la < end && *la != '\n'; ++la) {} \
    if (la < end) { ++la; } /* consume the '\n' itself */

/* Consume one character into c; go to end_loop if at EOF. */
#define lookAhead() \
    if (la >= end) { goto end_loop; } else { c = *la++; }

/* Emit a token at byte position P with length L. */
#define PUT_TOKEN(K, P, L) do { \
    token = tokens + (*tokens_idx)++; \
    *token = (Token){.kind = (K), .src = src, .pos = (P), .leng = (L)}; \
} while (0)
# @

== Identifier scanner

Returns a pointer to the first byte past the end of the identifier that starts
at `p`.  If `p` does not begin with an identifier start character the function
returns `p` unchanged (zero-length result).

# <<identifier scanner>>=
static const char *lexer_getIdentifier(const char *p, const char *end)
{
    if (p >= end) return p;
    char c = *p;
    if (!c_is_ident_start) return p;
    for (const char *q = p + 1; q < end; ++q) {
        c = *q;
        if (!c_is_ident_cont) return q;
    }
    return end;
}
# @

== Main lexer function

The function signature is identical to the original C version so that existing
callers need no changes.  The two new parameters are `errors` / `errors_size` /
`errors_idx` for error accumulation.

# <<lexer run>>=
int64_t lexer_run(int64_t src, const char specialChar,
      const char *buffer, int64_t end_pos, int64_t *begin_pos,
      Token *tokens, int64_t tokens_size, int64_t *tokens_idx,
      int64_t *states, int64_t states_size, int64_t *states_idx,
      LexerError *errors, int64_t errors_size, int64_t *errors_idx)
{
    const char *p   = buffer + *begin_pos;
    const char *end = buffer + end_pos;
    const char *la  = p;
    char        c;
    bool        done = false;
    int64_t     token_pos;
    TokenType   tokenKind;
    Token      *token;

    /* Track (line, col) for error messages.  Both are 1-based. */
    int64_t cur_line = 1, cur_col = 1;

    /* Helper: advance cur_line/cur_col past the range [from, to). */
#define ADVANCE_POSITION(from, to) do { \
    for (const char *_q = (from); _q < (to); ++_q) { \
        if (*_q == '\n') { ++cur_line; cur_col = 1; } else { ++cur_col; } \
    } \
} while (0)

    do {
        if (p >= end) { goto end_loop; }

        /* Ensure room for at least 3 tokens (gap + token + one spare). */
        if (*tokens_idx + 3 > tokens_size) {
            *begin_pos = p - buffer;
            return kEndTokenBuf;
        }

        if (!in_state(Macro)) {
            /* Fast-scan: skip ahead to the next special character. */
            la = (const char *) memchr(p, specialChar, end - p);
            if (la == NULL) { goto end_loop; }
            p = la;
        }

        c  = *p;
        la = p + 1;

        switch (states[*states_idx]) {

        /* ------------------------------------------------------------------ */
        case sMacro: {
            switch (c) {
            case ' ': case '\t': case '\r': case '\n':
                for (; la < end; ++la) {
                    c = *la;
                    if (!c_is_space) break;
                }
                tokenKind = kSpace; goto got_token;
            case ',':
                tokenKind = kComma; goto got_token;
            case ')':
                tokenKind = kCloseParen; POP_STATE(); goto got_token;
            case '=':
                tokenKind = kEqual; goto got_token;
            default:
                if (c_is_ident_start) {
                    la = lexer_getIdentifier(p, end);
                    tokenKind = kIdent; goto got_token;
                }
                /* Any other character in Macro state: fall through to
                   sBlock handling where specialChar sequences are parsed. */
                break;
            }
        } /* fall through */

        /* ------------------------------------------------------------------ */
        case sBlock: {
            if (c != specialChar) break; /* advance p by 1, continue */

            /* We are on a special character.  Consume it and look ahead. */
            lookAhead(); /* c = char after special; la points one past it */

            switch (c) {
            case '{':
                tokenKind = kBlockOpen; PUSH_STATE(sBlock); goto got_token;

            case '}': {
                if (*states_idx == 0) {
                    /* Unmatched %} — report error, emit as text. */
                    push_error(errors, errors_idx, errors_size,
                               cur_line, cur_col - 1,
                               "Unmatched block close: no open block");
                    la = p; /* p still points at specialChar; la = p+2 */
                    tokenKind = kText; goto got_token;
                }
                tokenKind = kBlockClose; POP_STATE(); goto got_token;
            }

            case '/': {
                lookAhead();
                if (c == '/') {
                    ADVANCE_TO_EOL();
                    tokenKind = kLineComment; goto got_token;
                } else if (c == '*') {
                    tokenKind = kCommentOpen; PUSH_STATE(sComment); goto got_token;
                } else {
                    push_error(errors, errors_idx, errors_size,
                               cur_line, cur_col,
                               "Unexpected char after '%c/' in block", specialChar);
                    tokenKind = kText; goto got_token;
                }
                break;
            }

            case '-': {
                lookAhead();
                if (c == '-') {
                    ADVANCE_TO_EOL();
                    tokenKind = kLineComment; goto got_token;
                } else {
                    push_error(errors, errors_idx, errors_size,
                               cur_line, cur_col,
                               "Unexpected char after '%c-' in block", specialChar);
                    tokenKind = kText; goto got_token;
                }
                break;
            }

            case '#':
                ADVANCE_TO_EOL();
                tokenKind = kLineComment; goto got_token;

            default: {
                if (c == specialChar) {
                    /* %% → Special */
                    tokenKind = kSpecial; goto got_token;
                }
                if (c == '(') {
                    /* %(ident) → Var.  la already points past '('. */
                    const char *id_end = lexer_getIdentifier(la, end);
                    if (id_end == la) {
                        push_error(errors, errors_idx, errors_size,
                                   cur_line, cur_col,
                                   "Var missing identifier after '%c('", specialChar);
                        tokenKind = kText; goto got_token;
                    }
                    la = id_end;
                    lookAhead();
                    if (c == ')') {
                        tokenKind = kVar; goto got_token;
                    }
                    push_error(errors, errors_idx, errors_size,
                               cur_line, cur_col,
                               "Var missing closing ')'");
                    tokenKind = kText; goto got_token;
                }
                /* Possibly %ident{ / %ident} / %ident( */
                if (c_is_ident_start) {
                    /* Back la up: we consumed one char of the identifier. */
                    --la;
                }
                {
                    const char *id_start = la;
                    la = lexer_getIdentifier(la, end);
                    if (la == id_start) {
                        /* Not an identifier character — truly unrecognised. */
                        push_error(errors, errors_idx, errors_size,
                                   cur_line, cur_col,
                                   "Unrecognized char after '%c' in block",
                                   specialChar);
                        la = p + 1; /* emit just the special char as text */
                        tokenKind = kText; goto got_token;
                    }
                    if (la >= end) {
                        /* EOF after %ident with no trailing punctuation. */
                        push_error(errors, errors_idx, errors_size,
                                   cur_line, cur_col,
                                   "Unrecognized sequence after '%c' in block",
                                   specialChar);
                        tokenKind = kText; goto got_token;
                    }
                    lookAhead();
                    switch (c) {
                    case '{': tokenKind = kBlockOpen;  PUSH_STATE(sBlock); goto got_token;
                    case '}':
                        if (*states_idx == 0) {
                            push_error(errors, errors_idx, errors_size,
                                       cur_line, cur_col - 1,
                                       "Unmatched block close: no open block");
                            la = p; tokenKind = kText; goto got_token;
                        }
                        tokenKind = kBlockClose; POP_STATE(); goto got_token;
                    case '(': tokenKind = kMacro; PUSH_STATE(sMacro); goto got_token;
                    default:
                        push_error(errors, errors_idx, errors_size,
                                   cur_line, cur_col,
                                   "Unrecognized char after '%c<ident>' in block",
                                   specialChar);
                        tokenKind = kText; goto got_token;
                    }
                }
            } /* default */
            } /* switch (c) inside sBlock */
        } break;

        /* ------------------------------------------------------------------ */
        case sComment: {
            if (c != specialChar) break;
            lookAhead();
            switch (c) {
            case '/':
                lookAhead();
                if (c == '*') { tokenKind = kCommentOpen; PUSH_STATE(sComment); goto got_token; }
                break;
            case '*':
                lookAhead();
                if (c == '/') { tokenKind = kCommentClose; POP_STATE(); goto got_token; }
                break;
            default: break;
            }
        } break;

        } /* outer switch (states[*states_idx]) */

        ++p;
        continue;

    end_loop:
        done      = true;
        tokenKind = kEOF;

        /* ---- EOF fixups ---- */
        /* If we stopped inside a Comment, that comment was never closed. */
        if (in_state(Comment)) {
            push_error(errors, errors_idx, errors_size,
                       cur_line, cur_col,
                       "Unclosed comment at end of input");
        }
        /* If the very last byte was the special char, emit it as text. */
        if (p > buffer && *(p - 1) == specialChar) {
            /* The gap logic below will handle emitting it as kText. */
        }

        p  = end;
        la = end;

    got_token:
        token_pos = p - buffer;

        /* ---- Emit any gap between the previous token and this one ---- */
        int64_t lastend;
        if (*tokens_idx > 0) {
            token   = tokens + *tokens_idx - 1;
            lastend = token->pos + token->leng;
        } else {
            lastend = 0;
        }
        {
            const int64_t gap = token_pos - lastend;
            if (gap > 0) { PUT_TOKEN(kText, lastend, gap); }
        }

        /* ---- Backward whitespace extension for line and block comments ----
           When a line comment (or block comment open) immediately follows
           trailing whitespace, absorb that whitespace into the comment token.
           This keeps output lines clean.  The Rust lexer omits this step
           (it is handled by the evaluator), but it is preserved here for
           compatibility with the original C pipeline. */
        if ((*tokens_idx > 0) &&
            ((tokenKind == kLineComment) || (tokenKind == kCommentOpen)))
        {
            token = tokens + *tokens_idx - 1;
            if ((token->kind == kSpace) || (token->kind == kText)) {
                int64_t j = token->pos + token->leng - 1;
                for (; j >= token->pos; --j) {
                    char ch = buffer[j];
                    if (ch != ' ' && ch != '\t') break;
                }
                token->leng = j - token->pos + 1;
                token_pos   = j + 1;
                if (j < token->pos) {
                    /* Previous token was entirely whitespace — discard it. */
                    --(*tokens_idx);
                }
            }
        }

        if (!done) { PUT_TOKEN(tokenKind, token_pos, la - (buffer + token_pos)); }
        p = la;

    } while (!done);

    /* Emit final EOF token. */
    {
        token = tokens + (*tokens_idx)++;
        *token = (Token){.kind = kEOF, .src = src, .pos = end_pos, .leng = 0};
    }

#undef ADVANCE_POSITION
    return kOk;
}
